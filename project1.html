Description du Projet

CinéBook est une plateforme web de réservation de tickets de cinéma offrant une expérience utilisateur fluide, sécurisée et intuitive. L'application se divise en deux interfaces distinctes avec des fonctionnalités spécifiques pour les utilisateurs et les administrateurs.

Fonctionnalités Principales

Côté Utilisateur

  • Consultation des films disponibles : Navigation avec filtres par genre, date et horaire
  • Sélection des séances : Choix de séances spécifiques avec visualisation des places disponibles
  • Réservation de billets : Réservation directe en ligne avec sélection de places
  • Gestion des réservations : Consultation, modification et annulation des réservations
  • Paiement en ligne : Système de paiement sécurisé
  • Notifications : Alertes par email pour confirmations et rappels

Côté Administrateur

  • Gestion des films : CRUD complet (ajout, modification, suppression)
  • Gestion des salles : Configuration des capacités et dispositions
  • Programmation des séances : Création et modification des horaires
  • Suivi des réservations : Vue d'ensemble des réservations en temps réel
  • Gestion des utilisateurs : Administration des comptes clients
  • Reporting : Génération de rapports de ventes et fréquentation

Code Exemple - Entité Film

Classe Film.java (Entity JPA)

package com.cinebook.metier;

import javax.persistence.*;
import java.io.Serializable;
import java.util.List;

@Entity
@Table(name = "films")
@NamedQueries({
    @NamedQuery(name = "Film.findAll", query = "SELECT f FROM Film f"),
    @NamedQuery(name = "Film.findByGenre", query = "SELECT f FROM Film f WHERE f.genre = :genre"),
    @NamedQuery(name = "Film.findByTitre", query = "SELECT f FROM Film f WHERE f.titre LIKE :titre")
})
public class Film implements Serializable {
    
    private static final long serialVersionUID = 1L;
    
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    @Column(name = "film_id")
    private Long id;
    
    @Column(name = "titre", nullable = false, length = 255)
    private String titre;
    
    @Column(name = "realisateur", length = 150)
    private String realisateur;
    
    @Column(name = "acteurs", length = 500)
    private String acteurs;
    
    @Column(name = "duree")
    private Integer duree; // en minutes
    
    @Enumerated(EnumType.STRING)
    @Column(name = "genre")
    private GenreFilm genre;
    
    @Column(name = "description", length = 2000)
    private String description;
    
    @Column(name = "affiche_url")
    private String afficheUrl;
    
    @Column(name = "date_sortie")
    private java.sql.Date dateSortie;
    
    @Column(name = "note_presse")
    private Double notePresse;
    
    @Column(name = "age_minimum")
    private Integer ageMinimum;
    
    @OneToMany(mappedBy = "film", cascade = CascadeType.ALL, fetch = FetchType.LAZY)
    private List seances;
    
    // Constructeurs
    public Film() {}
    
    public Film(String titre, String realisateur, Integer duree, GenreFilm genre) {
        this.titre = titre;
        this.realisateur = realisateur;
        this.duree = duree;
        this.genre = genre;
    }
    
    // Getters et Setters
    public Long getId() { return id; }
    public void setId(Long id) { this.id = id; }
    
    public String getTitre() { return titre; }
    public void setTitre(String titre) { this.titre = titre; }
    
    public String getRealisateur() { return realisateur; }
    public void setRealisateur(String realisateur) { this.realisateur = realisateur; }
    
    public String getActeurs() { return acteurs; }
    public void setActeurs(String acteurs) { this.acteurs = acteurs; }
    
    public Integer getDuree() { return duree; }
    public void setDuree(Integer duree) { this.duree = duree; }
    
    public GenreFilm getGenre() { return genre; }
    public void setGenre(GenreFilm genre) { this.genre = genre; }
    
    public String getDescription() { return description; }
    public void setDescription(String description) { this.description = description; }
    
    public String getAfficheUrl() { return afficheUrl; }
    public void setAfficheUrl(String afficheUrl) { this.afficheUrl = afficheUrl; }
    
    public java.sql.Date getDateSortie() { return dateSortie; }
    public void setDateSortie(java.sql.Date dateSortie) { this.dateSortie = dateSortie; }
    
    public Double getNotePresse() { return notePresse; }
    public void setNotePresse(Double notePresse) { this.notePresse = notePresse; }
    
    public Integer getAgeMinimum() { return ageMinimum; }
    public void setAgeMinimum(Integer ageMinimum) { this.ageMinimum = ageMinimum; }
    
    public List getSeances() { return seances; }
    public void setSeances(List seances) { this.seances = seances; }
    
    // Méthodes utilitaires
    public String getDureeFormatee() {
        if (duree == null) return "N/A";
        int heures = duree / 60;
        int minutes = duree % 60;
        return heures + "h" + (minutes < 10 ? "0" : "") + minutes;
    }
    
    @Override
    public String toString() {
        return "Film{" +
                "id=" + id +
                ", titre='" + titre + '\'' +
                ", realisateur='" + realisateur + '\'' +
                ", duree=" + duree +
                ", genre=" + genre +
                '}';
    }
    
    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (!(o instanceof Film)) return false;
        Film film = (Film) o;
        return id != null && id.equals(film.id);
    }
    
    @Override
    public int hashCode() {
        return getClass().hashCode();
    }
}

// Enum pour les genres de films
enum GenreFilm {
    ACTION,
    COMEDIE,
    DRAME,
    SCIENCE_FICTION,
    HORREUR,
    ROMANCE,
    ANIMATION,
    DOCUMENTAIRE,
    THRILLER,
    AVENTURE
}

Code Exemple - FilmController REST

FilmController.java (REST API)

package com.cinebook.controller;

import com.cinebook.metier.Film;
import com.cinebook.service.FilmService;
import javax.ws.rs.*;
import javax.ws.rs.core.MediaType;
import javax.ws.rs.core.Response;
import java.util.List;

@Path("/films")
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
public class FilmController {
    
    private final FilmService filmService = new FilmService();
    
    @GET
    @Path("/all")
    public Response getAllFilms() {
        try {
            List films = filmService.getAllFilms();
            return Response.ok(films).build();
        } catch (Exception e) {
            return Response.status(Response.Status.INTERNAL_SERVER_ERROR)
                    .entity("Erreur lors de la récupération des films: " + e.getMessage())
                    .build();
        }
    }
    
    @GET
    @Path("/{id}")
    public Response getFilmById(@PathParam("id") Long id) {
        try {
            Film film = filmService.getFilmById(id);
            if (film == null) {
                return Response.status(Response.Status.NOT_FOUND)
                        .entity("Film non trouvé avec l'ID: " + id)
                        .build();
            }
            return Response.ok(film).build();
        } catch (Exception e) {
            return Response.status(Response.Status.INTERNAL_SERVER_ERROR)
                    .entity("Erreur lors de la récupération du film: " + e.getMessage())
                    .build();
        }
    }
    
    @POST
    @Path("/create")
    public Response createFilm(Film film) {
        try {
            // Validation des données
            if (film.getTitre() == null || film.getTitre().trim().isEmpty()) {
                return Response.status(Response.Status.BAD_REQUEST)
                        .entity("Le titre du film est obligatoire")
                        .build();
            }
            
            if (film.getDuree() == null || film.getDuree() <= 0) {
                return Response.status(Response.Status.BAD_REQUEST)
                        .entity("La durée du film doit être positive")
                        .build();
            }
            
            Film createdFilm = filmService.createFilm(film);
            return Response.status(Response.Status.CREATED)
                    .entity(createdFilm)
                    .build();
        } catch (Exception e) {
            return Response.status(Response.Status.INTERNAL_SERVER_ERROR)
                    .entity("Erreur lors de la création du film: " + e.getMessage())
                    .build();
        }
    }
    
    @PUT
    @Path("/update/{id}")
    public Response updateFilm(@PathParam("id") Long id, Film film) {
        try {
            film.setId(id); // S'assurer que l'ID est correct
            Film updatedFilm = filmService.updateFilm(film);
            
            if (updatedFilm == null) {
                return Response.status(Response.Status.NOT_FOUND)
                        .entity("Film non trouvé avec l'ID: " + id)
                        .build();
            }
            
            return Response.ok(updatedFilm).build();
        } catch (Exception e) {
            return Response.status(Response.Status.INTERNAL_SERVER_ERROR)
                    .entity("Erreur lors de la mise à jour du film: " + e.getMessage())
                    .build();
        }
    }
    
    @DELETE
    @Path("/delete/{id}")
    public Response deleteFilm(@PathParam("id") Long id) {
        try {
            boolean deleted = filmService.deleteFilm(id);
            
            if (!deleted) {
                return Response.status(Response.Status.NOT_FOUND)
                        .entity("Film non trouvé avec l'ID: " + id)
                        .build();
            }
            
            return Response.ok().entity("Film supprimé avec succès").build();
        } catch (Exception e) {
            return Response.status(Response.Status.INTERNAL_SERVER_ERROR)
                    .entity("Erreur lors de la suppression du film: " + e.getMessage())
                    .build();
        }
    }
    
    @GET
    @Path("/search")
    public Response searchFilms(
            @QueryParam("titre") String titre,
            @QueryParam("genre") String genre,
            @QueryParam("realisateur") String realisateur) {
        
        try {
            List films = filmService.searchFilms(titre, genre, realisateur);
            return Response.ok(films).build();
        } catch (Exception e) {
            return Response.status(Response.Status.INTERNAL_SERVER_ERROR)
                    .entity("Erreur lors de la recherche: " + e.getMessage())
                    .build();
        }
    }
    
    @GET
    @Path("/now-playing")
    public Response getNowPlayingFilms() {
        try {
            List films = filmService.getNowPlayingFilms();
            return Response.ok(films).build();
        } catch (Exception e) {
            return Response.status(Response.Status.INTERNAL_SERVER_ERROR)
                    .entity("Erreur lors de la récupération des films à l'affiche: " + e.getMessage())
                    .build();
        }
    }
    
    @GET
    @Path("/coming-soon")
    public Response getComingSoonFilms() {
        try {
            List films = filmService.getComingSoonFilms();
            return Response.ok(films).build();
        } catch (Exception e) {
            return Response.status(Response.Status.INTERNAL_SERVER_ERROR)
                    .entity("Erreur lors de la récupération des films à venir: " + e.getMessage())
                    .build();
        }
    }
}

FilmServlet.java (Servlet Classique)

package com.cinebook.service;

import com.cinebook.metier.Film;
import com.cinebook.dao.FilmDao;
import com.cinebook.dao.impl.ImplFilm;
import javax.servlet.*;
import javax.servlet.http.*;
import javax.servlet.annotation.*;
import java.io.IOException;
import java.util.List;

@WebServlet(name = "FilmServlet", urlPatterns = {
    "/films",
    "/films/add",
    "/films/edit",
    "/films/delete",
    "/films/details"
})
@MultipartConfig(
    fileSizeThreshold = 1024 * 1024 * 2, // 2MB
    maxFileSize = 1024 * 1024 * 10,      // 10MB
    maxRequestSize = 1024 * 1024 * 50    // 50MB
)
public class FilmServlet extends HttpServlet {
    
    private FilmDao filmDao;
    
    @Override
    public void init() throws ServletException {
        super.init();
        filmDao = new ImplFilm(); // Injection de dépendance
    }
    
    @Override
    protected void doGet(HttpServletRequest request, HttpServletResponse response) 
            throws ServletException, IOException {
        
        String action = request.getServletPath();
        
        try {
            switch (action) {
                case "/films":
                    listFilms(request, response);
                    break;
                case "/films/add":
                    showAddForm(request, response);
                    break;
                case "/films/edit":
                    showEditForm(request, response);
                    break;
                case "/films/delete":
                    deleteFilm(request, response);
                    break;
                case "/films/details":
                    showFilmDetails(request, response);
                    break;
                default:
                    listFilms(request, response);
                    break;
            }
        } catch (Exception e) {
            throw new ServletException(e);
        }
    }
    
    @Override
    protected void doPost(HttpServletRequest request, HttpServletResponse response) 
            throws ServletException, IOException {
        
        String action = request.getServletPath();
        
        try {
            switch (action) {
                case "/films/add":
                    addFilm(request, response);
                    break;
                case "/films/edit":
                    updateFilm(request, response);
                    break;
                default:
                    listFilms(request, response);
                    break;
            }
        } catch (Exception e) {
            throw new ServletException(e);
        }
    }
    
    private void listFilms(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        
        try {
            // Récupération des paramètres de filtrage
            String genre = request.getParameter("genre");
            String searchTerm = request.getParameter("search");
            
            List films;
            
            if (genre != null && !genre.isEmpty()) {
                films = filmDao.findByGenre(genre);
            } else if (searchTerm != null && !searchTerm.isEmpty()) {
                films = filmDao.searchByTitle(searchTerm);
            } else {
                films = filmDao.findAll();
            }
            
            // Pagination
            int page = 1;
            int recordsPerPage = 10;
            
            if (request.getParameter("page") != null) {
                page = Integer.parseInt(request.getParameter("page"));
            }
            
            int start = (page - 1) * recordsPerPage;
            int end = Math.min(start + recordsPerPage, films.size());
            
            List filmsPage = films.subList(start, end);
            int numberOfPages = (int) Math.ceil(films.size() * 1.0 / recordsPerPage);
            
            // Ajout des attributs à la requête
            request.setAttribute("films", filmsPage);
            request.setAttribute("currentPage", page);
            request.setAttribute("numberOfPages", numberOfPages);
            request.setAttribute("totalFilms", films.size());
            request.setAttribute("selectedGenre", genre);
            request.setAttribute("searchTerm", searchTerm);
            
            // Forward vers la page JSP
            RequestDispatcher dispatcher = request.getRequestDispatcher("/WEB-INF/views/films/list.jsp");
            dispatcher.forward(request, response);
            
        } catch (Exception e) {
            request.setAttribute("errorMessage", "Erreur lors de la récupération des films: " + e.getMessage());
            request.getRequestDispatcher("/WEB-INF/views/error.jsp").forward(request, response);
        }
    }
    
    private void showAddForm(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        
        RequestDispatcher dispatcher = request.getRequestDispatcher("/WEB-INF/views/films/add.jsp");
        dispatcher.forward(request, response);
    }
    
    private void showEditForm(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        
        try {
            Long filmId = Long.parseLong(request.getParameter("id"));
            Film film = filmDao.findById(filmId);
            
            if (film == null) {
                request.setAttribute("errorMessage", "Film non trouvé avec l'ID: " + filmId);
                request.getRequestDispatcher("/WEB-INF/views/error.jsp").forward(request, response);
                return;
            }
            
            request.setAttribute("film", film);
            RequestDispatcher dispatcher = request.getRequestDispatcher("/WEB-INF/views/films/edit.jsp");
            dispatcher.forward(request, response);
            
        } catch (NumberFormatException e) {
            request.setAttribute("errorMessage", "ID de film invalide");
            request.getRequestDispatcher("/WEB-INF/views/error.jsp").forward(request, response);
        }
    }
    
    private void showFilmDetails(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        
        try {
            Long filmId = Long.parseLong(request.getParameter("id"));
            Film film = filmDao.findByIdWithSeances(filmId);
            
            if (film == null) {
                request.setAttribute("errorMessage", "Film non trouvé avec l'ID: " + filmId);
                request.getRequestDispatcher("/WEB-INF/views/error.jsp").forward(request, response);
                return;
            }
            
            request.setAttribute("film", film);
            RequestDispatcher dispatcher = request.getRequestDispatcher("/WEB-INF/views/films/details.jsp");
            dispatcher.forward(request, response);
            
        } catch (NumberFormatException e) {
            request.setAttribute("errorMessage", "ID de film invalide");
            request.getRequestDispatcher("/WEB-INF/views/error.jsp").forward(request, response);
        }
    }
    
    private void addFilm(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        
        try {
            // Récupération des paramètres du formulaire
            String titre = request.getParameter("titre");
            String realisateur = request.getParameter("realisateur");
            String acteurs = request.getParameter("acteurs");
            String dureeStr = request.getParameter("duree");
            String genre = request.getParameter("genre");
            String description = request.getParameter("description");
            
            // Validation
            if (titre == null || titre.trim().isEmpty()) {
                request.setAttribute("errorMessage", "Le titre est obligatoire");
                request.getRequestDispatcher("/WEB-INF/views/films/add.jsp").forward(request, response);
                return;
            }
            
            // Création de l'objet Film
            Film film = new Film();
            film.setTitre(titre);
            film.setRealisateur(realisateur);
            film.setActeurs(acteurs);
            
            if (dureeStr != null && !dureeStr.isEmpty()) {
                film.setDuree(Integer.parseInt(dureeStr));
            }
            
            film.setGenre(GenreFilm.valueOf(genre));
            film.setDescription(description);
            
            // Gestion de l'upload de l'affiche
            Part filePart = request.getPart("affiche");
            if (filePart != null && filePart.getSize() > 0) {
                String fileName = getFileName(filePart);
                String uploadPath = getServletContext().getRealPath("") + "uploads";
                
                File uploadDir = new File(uploadPath);
                if (!uploadDir.exists()) {
                    uploadDir.mkdir();
                }
                
                String filePath = uploadPath + File.separator + fileName;
                filePart.write(filePath);
                
                film.setAfficheUrl("uploads/" + fileName);
            }
            
            // Persistance
            Film createdFilm = filmDao.create(film);
            
            // Redirection avec message de succès
            request.getSession().setAttribute("successMessage", 
                "Film '" + createdFilm.getTitre() + "' ajouté avec succès!");
            response.sendRedirect(request.getContextPath() + "/films");
            
        } catch (Exception e) {
            request.setAttribute("errorMessage", "Erreur lors de l'ajout du film: " + e.getMessage());
            request.getRequestDispatcher("/WEB-INF/views/films/add.jsp").forward(request, response);
        }
    }
    
    private void updateFilm(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        
        try {
            Long filmId = Long.parseLong(request.getParameter("id"));
            Film existingFilm = filmDao.findById(filmId);
            
            if (existingFilm == null) {
                request.setAttribute("errorMessage", "Film non trouvé");
                request.getRequestDispatcher("/WEB-INF/views/error.jsp").forward(request, response);
                return;
            }
            
            // Mise à jour des attributs
            existingFilm.setTitre(request.getParameter("titre"));
            existingFilm.setRealisateur(request.getParameter("realisateur"));
            existingFilm.setActeurs(request.getParameter("acteurs"));
            existingFilm.setDescription(request.getParameter("description"));
            
            String dureeStr = request.getParameter("duree");
            if (dureeStr != null && !dureeStr.isEmpty()) {
                existingFilm.setDuree(Integer.parseInt(dureeStr));
            }
            
            String genre = request.getParameter("genre");
            if (genre != null && !genre.isEmpty()) {
                existingFilm.setGenre(GenreFilm.valueOf(genre));
            }
            
            // Mise à jour en base
            Film updatedFilm = filmDao.update(existingFilm);
            
            request.getSession().setAttribute("successMessage", 
                "Film '" + updatedFilm.getTitre() + "' modifié avec succès!");
            response.sendRedirect(request.getContextPath() + "/films");
            
        } catch (Exception e) {
            request.setAttribute("errorMessage", "Erreur lors de la modification: " + e.getMessage());
            request.getRequestDispatcher("/WEB-INF/views/films/edit.jsp").forward(request, response);
        }
    }
    
    private void deleteFilm(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        
        try {
            Long filmId = Long.parseLong(request.getParameter("id"));
            boolean deleted = filmDao.delete(filmId);
            
            if (deleted) {
                request.getSession().setAttribute("successMessage", 
                    "Film supprimé avec succès!");
            } else {
                request.getSession().setAttribute("errorMessage", 
                    "Erreur lors de la suppression du film");
            }
            
            response.sendRedirect(request.getContextPath() + "/films");
            
        } catch (Exception e) {
            request.setAttribute("errorMessage", "Erreur lors de la suppression: " + e.getMessage());
            request.getRequestDispatcher("/WEB-INF/views/films/list.jsp").forward(request, response);
        }
    }
    
    private String getFileName(Part part) {
        String contentDisp = part.getHeader("content-disposition");
        String[] tokens = contentDisp.split(";");
        for (String token : tokens) {
            if (token.trim().startsWith("filename")) {
                return token.substring(token.indexOf("=") + 2, token.length() - 1);
            }
        }
        return "";
    }
}

Script de création de la base

-- Base de données CinéBook
CREATE DATABASE IF NOT EXISTS cinebook_db;
USE cinebook_db;

-- Table des utilisateurs
CREATE TABLE users (
    user_id INT PRIMARY KEY AUTO_INCREMENT,
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    nom VARCHAR(100) NOT NULL,
    prenom VARCHAR(100) NOT NULL,
    telephone VARCHAR(20),
    role ENUM('CLIENT', 'ADMIN') DEFAULT 'CLIENT',
    date_inscription DATETIME DEFAULT CURRENT_TIMESTAMP,
    last_login DATETIME,
    is_active BOOLEAN DEFAULT TRUE,
    INDEX idx_email (email),
    INDEX idx_role (role)
);

-- Table des films
CREATE TABLE films (
    film_id INT PRIMARY KEY AUTO_INCREMENT,
    titre VARCHAR(255) NOT NULL,
    realisateur VARCHAR(150),
    acteurs TEXT,
    duree INT, -- en minutes
    genre ENUM('ACTION', 'COMEDIE', 'DRAME', 'SCIENCE_FICTION', 
               'HORREUR', 'ROMANCE', 'ANIMATION', 'DOCUMENTAIRE',
               'THRILLER', 'AVENTURE'),
    description TEXT,
    affiche_url VARCHAR(500),
    date_sortie DATE,
    note_presse DECIMAL(3,1),
    age_minimum INT DEFAULT 0,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    INDEX idx_genre (genre),

-- Vues utiles
CREATE VIEW vue_films_a_l_affiche AS
SELECT f.*, COUNT(s.seance_id) as nombre_seances
FROM films f
LEFT JOIN seances s ON f.film_id = s.film_id 
    AND s.date_heure >= CURDATE()
GROUP BY f.film_id
HAVING nombre_seances > 0;

CREATE VIEW vue_reservations_details AS
SELECT r.*, 
       u.nom, u.prenom, u.email,
       f.titre as film_titre,
       s.date_heure as date_seance,
       sa.numero as salle_numero,
       p.montant, p.type_paiement, p.statut as statut_paiement
FROM reservations r
JOIN users u ON r.user_id = u.user_id
JOIN seances s ON r.seance_id = s.seance_id
JOIN films f ON s.film_id = f.film_id
JOIN salles sa ON s.salle_id = sa.salle_id
LEFT JOIN paiements p ON r.reservation_id = p.reservation_id;

DELIMITER //
CREATE PROCEDURE sp_creer_reservation(
    IN p_user_id INT,
    IN p_seance_id INT,
    IN p_places_ids TEXT
)
BEGIN
    DECLARE v_prix_total DECIMAL(8,2) DEFAULT 0;
    DECLARE v_nombre_places INT DEFAULT 0;
    DECLARE v_code_reservation VARCHAR(20);
    
    -- Calcul du prix total
    SELECT SUM(prix_supplement + (SELECT prix FROM seances WHERE seance_id = p_seance_id))
    INTO v_prix_total
    FROM places
    WHERE FIND_IN_SET(place_id, p_places_ids) > 0;
    
    -- Nombre de places
    SET v_nombre_places = LENGTH(p_places_ids) - LENGTH(REPLACE(p_places_ids, ',', '')) + 1;
    
    -- Génération d'un code unique
    SET v_code_reservation = UPPER(CONCAT('CINE', DATE_FORMAT(NOW(), '%y%m%d'), 
                                        LPAD(FLOOR(RAND() * 10000), 4, '0')));
    
    -- Insertion de la réservation
    INSERT INTO reservations (user_id, seance_id, nombre_places, prix_total, code_reservation)
    VALUES (p_user_id, p_seance_id, v_nombre_places, v_prix_total, v_code_reservation);
    
    SET @last_reservation_id = LAST_INSERT_ID();
    
    -- Association des places
    INSERT INTO reservation_places (reservation_id, place_id)
    SELECT @last_reservation_id, place_id
    FROM places
    WHERE FIND_IN_SET(place_id, p_places_ids) > 0;
    
    -- Mise à jour de la disponibilité des places
    UPDATE places 
    SET is_disponible = FALSE 
    WHERE FIND_IN_SET(place_id, p_places_ids) > 0;
    
    -- Mise à jour des places disponibles dans la séance
    UPDATE seances 
    SET places_disponibles = places_disponibles - v_nombre_places 
    WHERE seance_id = p_seance_id;
    
    SELECT @last_reservation_id as reservation_id, v_code_reservation as code_reservation;
END//
DELIMITER ;
